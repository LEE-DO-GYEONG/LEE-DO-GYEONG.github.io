    return str(height) + 'cm'

def capitalize(lang):
    if pd.notnull(lang):  # NaN 이 아닌지
        return lang.capitalize()  # 첫 글자는 대문자로, 나머지는 소문자로
    return lang

df['SW특기'] = df['SW특기'].apply(capitalize)
df
